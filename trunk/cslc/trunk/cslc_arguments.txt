cslc CSL commands
--I <dirname>
--D 
--csl_maxerror <number> default = 30
--csl_maxwarnings <number> default = 30
--f foo.cf
--config_file config_file.xml
--csl_pp on/off/only 
  default = run
  on - run the CSL pp
  off- do not run the CSL pp
  only - only run the CSL pp and stop after running the CSL pp
--csl_pp_filename <filename> // create a single file for each CSL file processed by the CSL pp
cslc Verilog commands
--verilog_pp on/off/only 
  default = run
  on - run the Verilog pp
  off- do not run the Verilog pp
  only - only run the Verilog pp and stop after running the Verilog pp
--verilog_pp_filename <filename> // create a single file for each Verilog file processed by the Verilog pp
